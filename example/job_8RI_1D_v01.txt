  then

    echo "Folder does not exist"

epsilon_DMI=0 # epistasis of a single DMI in the double heterozygote (assuming codominance)
skip_mating_choice=1 # 0 for mate choice, 1 to turn it off

	#echo "exists"

	then
		echo "name_FA_file existed - skipped"
	else	
		echo "$name_FA_file"
		
		seed=$(python3 -c 'import random as R; print(R.randint(1, 2**32-1))')
		if test -f $parameter_file
		then		
			echo "parameter file already exists"
		else 
			echo -e "*\nseed $seed \npop_size $pop_size \nnb_loci $nb_loci \ndim $dim \nnb_gen $nb_gen \nmu $mu \nmu_mc $muMC \nmig $mig \nrho $rho \nrho_LB $rho_LB \nrho_CL $rho_CL \nsig $sig \nfit_alt $fit_alt \nepistasis_Fisher $epistasis_Fisher \nopt_one_dimI $opt_one_dimI \nopt_two_dimI $opt_two_dimI \nvar_pref $var_pref \nvar_cost $var_cost \nnb_dmi $nb_dmi \nepsilon_DMI $epsilon_DMI \nskip_mating_choice $skip_mating_choice \nnb_trials $nb_trials \nn_boot $n_boot \nlife_cycle $life_cycle \nload_mutation $load_mutation \n*" > ${parameter_file} # write the parameter file
			./../../main_version/Ri_measure ${parameter_file} ${mutation_file} ${name_pop_file} ${name_FA_file} # call for the program; output file has the same name than the folder plus an extra part corresponding to the recombination rate for easy identification. Note that since the parameters value are written in the file itself a mistake here, while not desirable is not critical.
			rm $parameter_file

	fi
